(function(){ const KEYS = { products:'qm_products_by_supplier_v1', intel:'qm_intel_prefill_v1', listing:'qm_listing_prefill_v1', crm:'qm_crm_v1', stores:'qm_stores_v1', activeStore:'qm_active_store_v1', margin:'qm_store_margin_pct' }; const safeParse = (value, fallback) => { try { return JSON.parse(value) ?? fallback; } catch(e){ return fallback; } }; const getStores = () => safeParse(localStorage.getItem(KEYS.stores), []); const setStores = (stores) => localStorage.setItem(KEYS.stores, JSON.stringify(stores)); const getMargin = () => Number(localStorage.getItem(KEYS.margin) || 35); function formatMoney(value){ return new Intl.NumberFormat('pl-PL',{style:'currency',currency:'PLN',maximumFractionDigits:2}).format(Number(value||0)); } function countPromoLeft(){ const stores = getStores(); return Math.max(0, 3 - stores.length); } function injectPromo(){ document.querySelectorAll('[data-promo-left]').forEach(el => { const left = countPromoLeft(); el.textContent = left > 0 ? `Promocja aktywna: jeszcze ${left} z 3 pierwszych sprzedawców wejdzie na start bez opłaty wdrożeniowej i z mocniejszym podbiciem na stronie głównej.` : 'Promocja dla pierwszych 3 sprzedawców została już wykorzystana. Kolejni wchodzą standardowo z pełnym dostępem do systemu.'; }); } function injectHeader(){ const slots = document.querySelectorAll('[data-shared-header]'); slots.forEach(slot => { slot.innerHTML = `
U Szefa menu główne • wejście do królestwa zarabiania
Start QualitetMarket Załóż sklep Reklama i rozwój Subskrypcje
`; }); } function fillDashboardStats(){ const stores = getStores(); const avgMargin = getMargin(); const salesBase = stores.length * 7800; const estProfit = salesBase * (avgMargin/100); document.querySelectorAll('[data-stores-count]').forEach(el => el.textContent = stores.length); document.querySelectorAll('[data-margin-default]').forEach(el => el.textContent = avgMargin + '%'); document.querySelectorAll('[data-est-sales]').forEach(el => el.textContent = formatMoney(salesBase)); document.querySelectorAll('[data-est-profit]').forEach(el => el.textContent = formatMoney(estProfit)); } function renderStores(targetSelector){ const target = document.querySelector(targetSelector); if(!target) return; const stores = getStores(); if(!stores.length){ target.innerHTML = `
Nie ma jeszcze zapisanych sklepów. Wejdź w generator, ustaw logo, wybierz motyw i zapisz pierwszy sklep.
`; return; } target.innerHTML = stores.map((store, idx) => `
${idx < 3 ? 'Promocja startowa' : 'Aktywny sklep'}

${store.name}

${store.niche || 'Sprzedaż wielobranżowa'} • plan: ${store.plan}

Logo: ${store.logoName || 'domyślne'} • motyw: ${store.theme}
`).join(''); } window.QM = { KEYS, safeParse, getStores, setStores, getMargin, formatMoney, injectPromo, injectHeader, fillDashboardStats, renderStores }; document.addEventListener('DOMContentLoaded', () => { injectHeader(); injectPromo(); fillDashboardStats(); renderStores('[data-stores-list]'); }); })();
