(function () { const KEYS = { stores: 'qm_stores_v1', activeStore: 'qm_active_store_v1', margin: 'qm_store_margin_pct', intel: 'qm_intel_prefill_v1', listing: 'qm_listing_prefill_v1', crm: 'qm_crm_v1', products: 'qm_products_by_supplier_v1' }; const safeParse = (value, fallback) => { try { return JSON.parse(value); } catch (e) { return fallback; } }; const getStores = () => safeParse(localStorage.getItem(KEYS.stores), []); const setStores = (stores) => localStorage.setItem(KEYS.stores, JSON.stringify(stores)); const getMargin = () => Number(localStorage.getItem(KEYS.margin) || 35); const setMargin = (value) => localStorage.setItem(KEYS.margin, String(value)); const fmt = (value) => new Intl.NumberFormat('pl-PL', {style: 'currency', currency: 'PLN'}).format(Number(value || 0)); function injectHeader() { document.querySelectorAll('[data-header]').forEach(el => { el.innerHTML = `
U Szefa QualitetMarket Załóż sklep Reklama i marża Subskrypcje
`; }); } function updatePromoText() { const left = Math.max(0, 3 - getStores().length); document.querySelectorAll('[data-promo-left]').forEach(el => { el.textContent = left > 0 ? `Promocja dla pierwszych 3 sprzedawców aktywna. Zostało ${left} miejsc.` : 'Promocja startowa została wykorzystana. System działa już w trybie standardowym.'; }); } function fillDashboard() { const stores = getStores(); const avgMargin = getMargin(); const salesBase = stores.length * 7800; const profit = salesBase * (avgMargin / 100); document.querySelectorAll('[data-stores-count]').forEach(el => el.textContent = stores.length); document.querySelectorAll('[data-margin-default]').forEach(el => el.textContent = avgMargin + '%'); document.querySelectorAll('[data-est-sales]').forEach(el => el.textContent = fmt(salesBase)); document.querySelectorAll('[data-est-profit]').forEach(el => el.textContent = fmt(profit)); } function renderStores(targetId) { const target = document.getElementById(targetId); if (!target) return; const stores = getStores(); if (!stores.length) { target.innerHTML = '
Nie ma jeszcze zapisanych sklepów. Wejdź do generatora sklepu i dodaj pierwszy sklep.
'; return; } target.innerHTML = stores.map((store, idx) => `
${idx < 3 ? 'Promocja startowa' : 'Aktywny sklep'}

${store.name}

${store.niche || 'Sprzedaż wielobranżowa'} • plan: ${store.plan}

Logo: ${store.logoName || 'domyślne'} • motyw: ${store.theme}

`).join(''); } document.addEventListener('DOMContentLoaded', () => { injectHeader(); updatePromoText(); fillDashboard(); renderStores('stores-list'); }); window.QM = { KEYS, safeParse, getStores, setStores, getMargin, setMargin, fmt, renderStores }; })();
