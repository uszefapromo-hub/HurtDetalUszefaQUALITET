(function () { function bindStoreGenerator() { const form = document.getElementById('store-generator-form'); if (!form || !window.QM) return; const chips = [...document.querySelectorAll('.theme-chip')]; const themeInput = document.getElementById('theme'); chips.forEach(chip => { chip.addEventListener('click', () => { chips.forEach(c => c.classList.remove('active')); chip.classList.add('active'); themeInput.value = chip.dataset.theme; }); }); form.addEventListener('submit', (e) => { e.preventDefault(); const fd = new FormData(form); const stores = QM.getStores(); const store = { id: Date.now(), name: fd.get('storeName'), niche: fd.get('niche'), plan: fd.get('plan'), theme: fd.get('theme'), logoName: (fd.get('logo') && fd.get('logo').name) ? fd.get('logo').name : 'logo-wlasne', createdAt: new Date().toISOString() }; stores.push(store); QM.setStores(stores); localStorage.setItem(QM.KEYS.activeStore, String(store.id)); localStorage.setItem(QM.KEYS.margin, String(fd.get('margin') || 35)); localStorage.setItem(QM.KEYS.intel, JSON.stringify({ storeName: store.name, niche: store.niche })); localStorage.setItem(QM.KEYS.listing, JSON.stringify({ theme: store.theme, plan: store.plan })); localStorage.setItem(QM.KEYS.crm, JSON.stringify({ ownerPanel: true, storeId: store.id })); const box = document.getElementById('save-result'); if (box) { box.innerHTML = `
Sklep zapisany. Nazwa: ${store.name}. Plan: ${store.plan}. Motyw: ${store.theme}.
`; } QM.renderStores('stores-list'); form.reset(); }); } document.addEventListener('DOMContentLoaded', bindStoreGenerator); })();
